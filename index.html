// ========== LOCAL OFFERING CODES MANAGEMENT ========== window.openLocalOfferingCodesModal = function() { const semesterData = getCurrentSemesterData(); const modal = document.getElementById('localOfferingCodesModal'); const content = document.getElementById('localOfferingCodesModalContent'); if (!semesterData.courses || semesterData.courses.length === 0) { content.innerHTML = '

Please add courses first in the Data Hub.

'; modal.classList.add('active'); return; } // Sort courses and colleges for dropdowns const sortedCourses = [...semesterData.courses].sort((a, b) => (a.Code || '').localeCompare(b.Code || '')); const sortedColleges = [...semesterData.colleges].sort(); let html = `
📍 Local Offering Codes
Map official course codes to college-specific local offering codes. Select a course and college to filter, then edit directly in the table.
Course Code Course Name College Local Offering Code

No matching offering codes found. Try different filters.

0 entries | Unsaved changes: 0
`; content.innerHTML = html; // Initial render with all codes window.filterLocalOfferingCodes(); modal.classList.add('active'); }; window.filterLocalOfferingCodes = function() { const semesterData = getCurrentSemesterData(); const courseFilter = document.getElementById('locCodeCourseFilter')?.value || 'ALL'; const collegeFilter = document.getElementById('locCodeCollegeFilter')?.value || 'ALL'; const sortedCourses = [...semesterData.courses].sort((a, b) => (a.Code || '').localeCompare(b.Code || '')); const sortedColleges = [...semesterData.colleges].sort(); const tableBody = document.getElementById('localOfferingCodesTableBody'); const emptyState = document.getElementById('locCodeEmptyState'); if (!tableBody) return; let html = ''; let count = 0; // Filter courses const filteredCourses = courseFilter === 'ALL' ? sortedCourses : sortedCourses.filter(c => c.Code === courseFilter); const filteredColleges = collegeFilter === 'ALL' ? sortedColleges : sortedColleges.filter(c => c === collegeFilter); filteredCourses.forEach(course => { filteredColleges.forEach(college => { const key = `${course.Code}|${college}`; const localCode = semesterData.localOfferingCodes[key] || ''; count++; html += ` ${course.Code} ${course.Name} ${college} `; }); }); tableBody.innerHTML = html; // Update count const countEl = document.getElementById('locCodeCount'); const unsavedEl = document.getElementById('locCodeUnsavedCount'); if (countEl) countEl.textContent = count; if (unsavedEl) unsavedEl.textContent = count; // Show/hide empty state if (emptyState) { emptyState.classList.toggle('hidden', count > 0); } // Add Enter key support document.querySelectorAll('.loc-code-input').forEach(input => { input.addEventListener('keypress', (e) => { if (e.key === 'Enter') { e.preventDefault(); // Move to next input or save all const inputs = Array.from(document.querySelectorAll('.loc-code-input')); const currentIndex = inputs.indexOf(e.target); if (currentIndex < inputs.length - 1) { inputs[currentIndex + 1].focus(); } else { window.saveAllLocalOfferingCodes(); } } }); }); }; window.markLocalCodeUnsaved = function(input) { // Visual feedback for unsaved changes input.style.borderColor = '#f59e0b'; input.style.backgroundColor = '#fffbeb'; input.classList.add('unsaved'); }; window.saveAllLocalOfferingCodes = function() { const semesterData = getCurrentSemesterData(); const inputs = document.querySelectorAll('.loc-code-input'); let savedCount = 0; let clearedCount = 0; inputs.forEach(input => { const key = input.dataset.key; const value = input.value.trim(); if (value) { semesterData.localOfferingCodes[key] = value; savedCount++; } else { delete semesterData.localOfferingCodes[key]; clearedCount++; } // Visual feedback for saved input.style.borderColor = '#10b981'; input.style.backgroundColor = '#ecfdf5'; input.classList.remove('unsaved'); }); saveData(); // Update unsaved count const unsavedEl = document.getElementById('locCodeUnsavedCount'); if (unsavedEl) unsavedEl.textContent = '0'; alert(`✅ Saved ${savedCount} local offering codes! (Cleared ${clearedCount} empty entries)`); }; window.clearAllLocalCodes = function() { if (confirm('⚠️ Delete ALL local offering codes? This cannot be undone.')) { const semesterData = getCurrentSemesterData(); semesterData.localOfferingCodes = {}; saveData(); // Clear all inputs document.querySelectorAll('.loc-code-input').forEach(input => { input.value = ''; input.style.borderColor = ''; input.style.backgroundColor = ''; }); const unsavedEl = document.getElementById('locCodeUnsavedCount'); if (unsavedEl) unsavedEl.textContent = '0'; alert('✅ All local offering codes cleared!'); } }; window.getLocalOfferingCode = function(courseCode, college) { const semesterData = getCurrentSemesterData(); const key = `${courseCode}|${college}`; return semesterData.localOfferingCodes[key] || ''; }; // ========== END LOCAL OFFERING CODES ==========
